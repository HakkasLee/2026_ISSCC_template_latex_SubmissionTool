% - Second paragraph:
%     * Space before: 9.15 pt, space after: 0 pt
%     * Line spacing: 1.08

\documentclass[12pt]{article}

% ==== Geometry settings ====
\usepackage[paper=letterpaper,
            top=2.4cm,
            bottom=0.49cm,
            left=2.54cm,
            right=2.54cm,
            includehead=false,
            includefoot=false,
            nomarginpar]{geometry}

\usepackage{setspace}
\setlength{\parindent}{0pt} % no indent

% ==== Fine-tune top offset ====
\makeatletter
\setlength{\topskip}{18pt}   % move text down (~6mm)
\setlength{\headheight}{0pt}
\setlength{\headsep}{0pt}
\setlength{\voffset}{0pt}
\makeatother

% ==== Font setup ====
\usepackage{fontspec} % enable system fonts
\newfontfamily\arialfont[Path=font/]{arial.ttf} % Arial regular
\newfontfamily\arialboldfont[Path=font/]{Arial-Bold.ttf} % Arial bold

% ==== Paragraph environments ====
\newenvironment{firstpara}{
  \vspace*{10pt}   % 10pt before
  \setstretch{1.0}
}{\vspace{0pt}}

\newenvironment{secondpara}{
  \vspace*{9.15pt} % 9.15pt before
  \setstretch{1.08}
}{\vspace{0pt}}

\begin{document}

\begin{firstpara}
{\arialboldfont % apply Arial-Bold only in this block
Abstract: 
}
\end{firstpara}

\begin{secondpara}
{\arialfont % apply Arial regular only in this block
Write your abstract here.
}
\end{secondpara}

\end{document}
